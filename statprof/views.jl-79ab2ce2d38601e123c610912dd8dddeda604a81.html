3 """
4 replace_ref_begin_end!(ex)
5
6 Recursively replace occurrences of the symbols `:begin` and `:end` in a "ref" expression
7 (i.e. `A[...]`) `ex` with the appropriate function calls (`firstindex` or `lastindex`).
8 Replacement uses the closest enclosing ref, so
9
10 A[B[end]]
11
12 should transform to
13
14 A[B[lastindex(B)]]
15
16 """
17 replace_ref_begin_end!(ex) = replace_ref_begin_end_!(ex, nothing)[1]
18 # replace_ref_begin_end_!(ex,withex) returns (new ex, whether withex was used)
19 function replace_ref_begin_end_!(ex, withex)
20 used_withex = false
21 if isa(ex,Symbol)
22 if ex === :begin
23 withex === nothing && error("Invalid use of begin")
24 return withex[1], true
25 elseif ex === :end
26 withex === nothing && error("Invalid use of end")
27 return withex[2], true
28 end
29 elseif isa(ex,Expr)
30 if ex.head === :ref
31 ex.args[1], used_withex = replace_ref_begin_end_!(ex.args[1], withex)
32 S = isa(ex.args[1],Symbol) ? ex.args[1]::Symbol : gensym(:S) # temp var to cache ex.args[1] if needed
33 used_S = false # whether we actually need S
34 # new :ref, so redefine withex
35 nargs = length(ex.args)-1
36 if nargs == 0
37 return ex, used_withex
38 elseif nargs == 1
39 # replace with lastindex(S)
40 ex.args[2], used_S = replace_ref_begin_end_!(ex.args[2], (:($firstindex($S)),:($lastindex($S))))
41 else
42 n = 1
43 J = lastindex(ex.args)
44 for j = 2:J
45 exj, used = replace_ref_begin_end_!(ex.args[j], (:($firstindex($S,$n)),:($lastindex($S,$n))))
46 used_S |= used
47 ex.args[j] = exj
48 if isa(exj,Expr) && exj.head === :...
49 # splatted object
50 exjs = exj.args[1]
51 n = :($n + length($exjs))
52 elseif isa(n, Expr)
53 # previous expression splatted
54 n = :($n + 1)
55 else
56 # an integer
57 n += 1
58 end
59 end
60 end
61 if used_S && S !== ex.args[1]
62 S0 = ex.args[1]
63 ex.args[1] = S
64 ex = Expr(:let, :($S = $S0), ex)
65 end
66 else
67 # recursive search
68 for i = eachindex(ex.args)
69 ex.args[i], used = replace_ref_begin_end_!(ex.args[i], withex)
70 used_withex |= used
71 end
72 end
73 end
74 ex, used_withex
75 end

77 """
78 @view A[inds...]
79
80 Transform the indexing expression `A[inds...]` into the equivalent [`view`](@ref) call.
81
82 This can only be applied directly to a single indexing expression and is particularly
83 helpful for expressions that include the special `begin` or `end` indexing syntaxes
84 like `A[begin, 2:end-1]` (as those are not supported by the normal [`view`](@ref)
85 function).
86
87 Note that `@view` cannot be used as the target of a regular assignment (e.g.,
88 `@view(A[1, 2:end]) = ...`), nor would the un-decorated
89 [indexed assignment](@ref man-indexed-assignment) (`A[1, 2:end] = ...`)
90 or broadcasted indexed assignment (`A[1, 2:end] .= ...`) make a copy. It can be useful,
91 however, for _updating_ broadcasted assignments like `@view(A[1, 2:end]) .+= 1`
92 because this is a simple syntax for `@view(A[1, 2:end]) .= @view(A[1, 2:end]) + 1`,
93 and the indexing expression on the right-hand side would otherwise make a
94 copy without the `@view`.
95
96 See also [`@views`](@ref) to switch an entire block of code to use views for non-scalar indexing.
97
98 !!! compat "Julia 1.5"
99 Using `begin` in an indexing expression to refer to the first index requires at least
100 Julia 1.5.
101
102 # Examples
103 ```jldoctest
104 julia> A = [1 2; 3 4]
105 2×2 Matrix{Int64}:
106 1 2
107 3 4
108
109 julia> b = @view A[:, 1]
110 2-element view(::Matrix{Int64}, :, 1) with eltype Int64:
111 1
112 3
113
114 julia> fill!(b, 0)
115 2-element view(::Matrix{Int64}, :, 1) with eltype Int64:
116 0
117 0
118
119 julia> A
120 2×2 Matrix{Int64}:
121 0 2
122 0 4
123 ```
124 """
125 macro view(ex)
126 if Meta.isexpr(ex, :ref)
127 ex = replace_ref_begin_end!(ex)
128 if Meta.isexpr(ex, :ref)
129 ex = Expr(:call, view, ex.args...)
130 else # ex replaced by let ...; foo[...]; end
131 @assert Meta.isexpr(ex, :let) && Meta.isexpr(ex.args[2], :ref)
132 ex.args[2] = Expr(:call, view, ex.args[2].args...)
133 end
134 Expr(:&&, true, esc(ex))
135 else
136 throw(ArgumentError("Invalid use of @view macro: argument must be a reference expression A[...]."))
137 end
138 end

151 # _views implements the transformation for the @views macro.
152 # @views calls esc(_views(...)) to work around #20241,
153 # so any function calls we insert (to maybeview, or to
154 # firstindex and lastindex in replace_ref_begin_end!) must be interpolated
155 # as values rather than as symbols to ensure that they are called
156 # from Base rather than from the caller's scope.
157 _views(x) = x
158 function _views(ex::Expr)
159 if ex.head in (:(=), :(.=))
160 # don't use view for ref on the lhs of an assignment,
161 # but still use views for the args of the ref:
162 arg1 = ex.args[1]
163 Expr(ex.head, Meta.isexpr(arg1, :ref) ?
164 Expr(:ref, mapany(_views, (arg1::Expr).args)...) : _views(arg1),
165 _views(ex.args[2]))
166 elseif ex.head === :ref
167 Expr(:call, maybeview, mapany(_views, ex.args)...)::Expr
168 else
169 h = string(ex.head)
170 # don't use view on the lhs of an op-assignment a[i...] += ...
171 if last(h) == '=' && Meta.isexpr(ex.args[1], :ref)
172 lhs = ex.args[1]::Expr
173
174 # temp vars to avoid recomputing a and i,
175 # which will be assigned in a let block:
176 a = gensym(:a)
177 i = let lhs=lhs # #15276
178 [gensym(:i) for k = 1:length(lhs.args)-1]
179 end
180
181 # for splatted indices like a[i, j...], we need to
182 # splat the corresponding temp var.
183 I = similar(i, Any)
184 for k = 1:length(i)
185 argk1 = lhs.args[k+1]
186 if Meta.isexpr(argk1, :...)
187 I[k] = Expr(:..., i[k])
188 lhs.args[k+1] = (argk1::Expr).args[1] # unsplat
189 else
190 I[k] = i[k]
191 end
192 end
193
194 Expr(:let,
195 Expr(:block,
196 :($a = $(_views(lhs.args[1]))),
197 Any[:($(i[k]) = $(_views(lhs.args[k+1]))) for k=1:length(i)]...),
198 Expr(first(h) == '.' ? :(.=) : :(=), :($a[$(I...)]),
199 Expr(:call, Symbol(h[1:end-1]),
200 :($maybeview($a, $(I...))),
201 mapany(_views, ex.args[2:end])...)))
202 else
203 exprarray(ex.head, mapany(_views, ex.args))
204 end
205 end
206 end

208 """
209 @views expression
210
211 Convert every array-slicing operation in the given expression
212 (which may be a `begin`/`end` block, loop, function, etc.)
213 to return a view. Scalar indices, non-array types, and
214 explicit [`getindex`](@ref) calls (as opposed to `array[...]`) are
215 unaffected.
216
217 !!! note
218 The `@views` macro only affects `array[...]` expressions
219 that appear explicitly in the given `expression`, not array slicing that
220 occurs in functions called by that code.
221
222 !!! compat "Julia 1.5"
223 Using `begin` in an indexing expression to refer to the first index requires at least
224 Julia 1.5.
225
226 # Examples
227 ```jldoctest
228 julia> A = zeros(3, 3);
229
230 julia> @views for row in 1:3
231 b = A[row, :]
232 b[:] .= row
233 end
234
235 julia> A
236 3×3 Matrix{Float64}:
237 1.0 1.0 1.0
238 2.0 2.0 2.0
239 3.0 3.0 3.0
240 ```
241 """
